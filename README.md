# Todo---Flask

# Create a Virtual Environment

python3 -m venv myapp

# Activate the environment
cd myapp/

cd Scripts

activate.bat

# Install Flask Dependencies
pip install flask

pip install flask_wtf

# Add app.py to your myapp directory
Copy app.py from Github to your myapp directory


# Create templates directory and import index.html from Github
mkdir templates
copy index.html from github to your /myapp/templates directory

flask run

The app should be available at http://127.0.0.1:5000/
